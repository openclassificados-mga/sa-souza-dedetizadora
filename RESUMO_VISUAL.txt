export const dadosCliente = {
  ┌─ CORES (customizáveis)
  │  ├─ primaryColor     (cor principal)
  │  ├─ secondaryColor   (cor secundária)
  │  └─ accentColor      (cor de destaque)
  │
  ├─ CONTATO
  │  ├─ whatsappNumber   (número com país e DDD)
  │  ├─ whatsappLink     (link https://wa.me/ automático)
  │  ├─ instagram        (perfil)
  │  ├─ email            (email de contato)
  │  ├─ endereco         (endereço físico)
  │  └─ telefone         (telefone formatado)
  │
  ├─ CONTEÚDO
  │  ├─ tituloPrincipal  (título do hero)
  │  ├─ subtitulo        (descrição com \n)
  │  └─ sobreNos         (texto sobre empresa)
  │
  ├─ SERVIÇOS (array)
  │  ├─ id               (identificador único)
  │  ├─ titulo           (nome do serviço)
  │  ├─ descricao        (descrição)
  │  ├─ icone            (emoji 🎯)
  │  └─ destaque         (opcional: true/false)
  │
  ├─ RECURSOS (array)
  │  ├─ titulo           (nome do recurso)
  │  └─ descricao        (breve descrição)
  │
  └─ SEO
     ├─ siteTitle        (título do site)
     ├─ description      (meta description)
     ├─ url              (URL do domínio)
     └─ ogImage          (imagem Open Graph)

┌─────────────────────────────────────────────────────────────────────────┐
│ 📊 ANTES vs DEPOIS                                                    │
└─────────────────────────────────────────────────────────────────────────┘

ANTES (Hardcoded)
├─ Dados em 4+ arquivos diferentes
├─ Cores espalhadas em classes Tailwind
├─ Difícil de customizar
├─ Não escalável
└─ Cada novo cliente = 4 arquivos a editar

DEPOIS (Parametrizado)
├─ Tudo em 1 arquivo (data/cliente.ts)
├─ Cores via CSS Variables
├─ Fácil de customizar
├─ Escalável para múltiplos clientes
└─ Cada novo cliente = 1 arquivo a editar ✅
